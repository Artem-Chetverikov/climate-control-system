DSsensor_Simpl

Библиотека взаимодействия с датчиком температуры DS18b20


Пример:
#include "DSsensor_Simpl.h"


OneWire oneWire_5(5);		// интерфейс OneWire на пине 5



//определяем экземпляр датчика (адрес,,,,,,,, интерфейс OneWire, точность результата( 9/10/11/12)):

DSsensor_Simpl 	 Temp_1(0x28, 0xFF, 0xD3, 0x59, 0x58, 0x16, 0x04, 0x6A, oneWire_5, 10);

Temp_1.Read();	//опрос датчика


t = Temp_1.ReturnVal();				
			
			

e = Temp_1.ReturnErr();		//вывод количества ошибок при опросе
