MHZ19_Simpl

Библиотека взаимодействия с датчиком температуры DS18b20


Пример:
#include "MHZ19_Simpl.h"



//определяем экземпляр датчика (используемый SPI):

MHZ19_Simpl 	 MHZ19_1_CO2(Serial2);





MHZ19_1_CO2.Read();	//опрос датчика


t = MHZ19_1_CO2.ReturnVal();				
			
			

e = MHZ19_1_CO2.ReturnErr();		//вывод количества ошибок при опросе
