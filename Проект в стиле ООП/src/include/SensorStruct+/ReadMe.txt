SensorStruct

Библиотека взаимодействия с датчиками и хранения результатов их опроса


Пример:
#include "SensorStruct.h"


SensorStr	 DHT_1(10);	//определяем датчик на пине №10

DHT_1.Read();	//опрос датчика


t = DHT_1.Temp();				
h = DHT_1.Hum();				
t1 = DHT_1.preTemp();			
h1 = DHT_1.preHum();			

DHT_1.Error();		//вывод количества ошибок при опросе
